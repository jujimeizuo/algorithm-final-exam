std::vector<int> Loading(const std::vector<int>& w, int V) {
    int n = w.size();
    std::vector<int> order(n);
    std::iota(order.begin(), order.end(), 0);
    std::sort(order.begin(), order.end(), [&w](int i, int j) {
        return w[i] < w[j];
    });
    std::vector<int> x(n);
    for (int i = 0; i < n && w[order[i]] <= c; i++) {
        x[w[order[i]]] = 1;
        c -= w[order[i]];
    }
    return x;
}
